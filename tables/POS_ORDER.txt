create table POS_ORDER
(
	ID NUMBER,
	DESCRIPTION VARCHAR2(200),
	PLACE_TIME DATE default NULL,
	SERVE_TIME DATE default NULL,
	LOGGED_USER VARCHAR2(30) default NULL,
	TOTAL_PRICE FLOAT,
	DISCOUNT FLOAT,
	CARD_NUMBER VARCHAR2(50)
)
/

CREATE SEQUENCE "POS_ORDER_SEQ"
  MINVALUE 100
  INCREMENT BY 1 START WITH 101
  CACHE 100;
/

create or replace trigger BEFORE_INSERT_POS_ORDER
  before insert
  on POS_ORDER
  for each row
  BEGIN
    :new.ID := POS_ORDER_SEQ.nextval;
  END;
