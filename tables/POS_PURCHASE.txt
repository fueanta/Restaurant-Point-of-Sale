create table POS_PURCHASE
(
	ID NUMBER,
	DESCRIPTION VARCHAR2(200),
	TIME DATE default NULL,
	LOGGED_USER VARCHAR2(30) default NULL,
	SUPP_ID NUMBER,
	TOTAL_COST FLOAT,
	DISCOUNT_RATE FLOAT
)
/

CREATE SEQUENCE "POS_PURCHASE_SEQ"
  MINVALUE 100
  INCREMENT BY 1 START WITH 101
  CACHE 100;
/

create or replace trigger BEFORE_INSERT_POS_PURCHASE
  before insert
  on POS_PURCHASE
  for each row
  BEGIN
    :new.ID := POS_PURCHASE_SEQ.nextval;
  END;
