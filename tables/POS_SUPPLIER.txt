create table POS_SUPPLIER
(
	ID NUMBER,
	NAME VARCHAR2(20),
	ADDRESS VARCHAR2(100),
	CONTACT_NUMBER VARCHAR2(20),
	EMAIL VARCHAR2(50),
	DETAILS VARCHAR2(200),
	DATE_ADDED DATE
)
/

CREATE SEQUENCE "SUPPLIER_SEQ"
  MINVALUE 1000
  INCREMENT BY 1 START WITH 1001
  CACHE 100;
/

create or replace trigger BEFORE_INSERT_SUPPLIER
  before insert
  on POS_SUPPLIER
  for each row
  BEGIN
    :new.ID := SUPPLIER_SEQ.nextval;
  END;
/
