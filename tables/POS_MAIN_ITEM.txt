create table POS_MAIN_ITEM
(
	ID NUMBER,
	NAME VARCHAR2(20),
	PRICE FLOAT,
	DESCRIPTION VARCHAR2(200),
	T_ID NUMBER,
	DISCOUNT_RATE FLOAT,
	AVAILABILITY CHAR,
	OTHER_CHARGES FLOAT,
	COUNT NUMBER,
	PICTURE BLOB
)
/

CREATE SEQUENCE "MAIN_ITEM_SEQ"
  MINVALUE 10
  INCREMENT BY 1 START WITH 11
  CACHE 100;
/

create or replace trigger BEFORE_INSERT_MAIN_ITEM
  before insert
  on POS_MAIN_ITEM
  for each row
  BEGIN
    :new.ID := MAIN_ITEM_SEQ.nextval;
  END;
